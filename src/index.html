quero um código para executar no console do browse que que obtenha o html de um elemento, a partir desse elemento quero buscar o css de cada classe usado dentro de cada elemento filho

Aqui está um código JavaScript que você pode executar no console do navegador para obter o HTML de um elemento específico e, em seguida, buscar os estilos CSS aplicados a cada classe dos elementos filhos: 

javascript
function getElementHtmlAndCss(selector) {
    const element = document.querySelector(selector);
    if (!element) {
        console.error("Elemento não encontrado");
        return;
    }

    // Função para obter os estilos CSS de uma classe
    function getCssOfClass(className) {
        const styles = {};
        for (let sheet of document.styleSheets) {
            for (let rule of sheet.cssRules) {
                if (rule.selectorText && rule.selectorText.includes(`.${className}`)) {
                    for (let style of rule.style) {
                        styles[style] = rule.style[style];
                    }
                }
            }
        }
        return styles;
    }

    // Obter HTML do elemento
    const elementHtml = element.outerHTML;
    console.log("HTML do elemento:", elementHtml);

    // Obter todas as classes dos elementos filhos
    const classes = new Set();
    element.querySelectorAll("*").forEach(child => {
        child.classList.forEach(cls => classes.add(cls));
    });

    // Obter os estilos CSS para cada classe
    const cssStyles = {};
    classes.forEach(cls => {
        cssStyles[cls] = getCssOfClass(cls);
    });

    console.log("Estilos CSS das classes:");
    console.log(cssStyles);

    return { html: elementHtml, css: cssStyles };
}

// Exemplo de uso
getElementHtmlAndCss("#seu-seletor-aqui");

Para usar esse código:

  1. Abra o navegador e vá para a página onde você deseja executar o script.
  2. Abra o console do navegador (normalmente, você pode abrir pressionando F12 ou Ctrl+Shift+I e indo para a aba "Console").
  3. Cole o código acima no console e pressione Enter.
  4. Substitua #seu-seletor-aqui pelo seletor do elemento desejado.

O script irá imprimir no console o HTML do elemento e os estilos CSS aplicados a cada classe dos elementos filhos.

quero que o output seja o html com o style contendo as definições do css

Entendi! Aqui está um código que gera o HTML do elemento com os estilos CSS aplicados inline:

function getElementHtmlWithInlineStyles(selector) {
    const element = document.querySelector(selector);
    if (!element) {
        console.error("Elemento não encontrado");
        return;
    }

    // Função para obter os estilos CSS de um elemento
    function getComputedStyleAsString(element) {
        const computedStyle = window.getComputedStyle(element);
        let styleString = "";
        for (let i = 0; i < computedStyle.length; i++) {
            const prop = computedStyle[i];
            styleString += `${prop}: ${computedStyle.getPropertyValue(prop)}; `;
        }
        return styleString;
    }

    // Função recursiva para aplicar estilos inline
    function applyInlineStyles(element) {
        const children = element.children;
        for (let i = 0; i < children.length; i++) {
            applyInlineStyles(children[i]);
        }
        element.setAttribute("style", getComputedStyleAsString(element));
    }

    // Clonar o elemento para não alterar o DOM original
    const clonedElement = element.cloneNode(true);
    applyInlineStyles(clonedElement);

    // Obter o HTML do elemento com estilos inline
    const elementHtmlWithStyles = clonedElement.outerHTML;
    console.log("HTML do elemento com estilos inline:");
    console.log(elementHtmlWithStyles);

    return elementHtmlWithStyles;
}

    function styleObjectToString(styleObj) {

    function applyInlineStyles(element) {

Quero converter o objeto css: cssStyles em uma definição de classes css para colar em um html

Para converter o objeto cssStyles em uma definição de classes CSS que pode ser colada em um HTML, você pode adicionar uma função que transforma o objeto em uma string CSS formatada. Aqui está o código atualizado: 

javascript
function getElementHtmlAndCss(selector) {
    const element = document.querySelector(selector);
    if (!element) {
        console.error("Elemento não encontrado");
        return;
    }

    // Função para obter os estilos CSS de uma classe
    function getCssOfClass(className) {
        const styles = {};
        for (let sheet of document.styleSheets) {
            for (let rule of sheet.cssRules) {
                if (rule.selectorText && rule.selectorText.includes(`.${className}`)) {
                    for (let style of rule.style) {
                        styles[style] = rule.style[style];
                    }
                }
            }
        }
        return styles;
    }

    // Função para converter objeto de estilos para string CSS
    function cssObjectToString(cssObject) {
        return Object.entries(cssObject).map(([prop, value]) => `${prop}: ${value};`).join(' ');
    }

    // Função para converter cssStyles para string CSS
    function cssStylesToString(cssStyles) {
        return Object.entries(cssStyles).map(([className, styles]) => {
            const styleString = cssObjectToString(styles);
            return `.${className} { ${styleString} }`;
        }).join('\n');
    }

    // Obter HTML do elemento
    const elementHtml = element.outerHTML;
    console.log("HTML do elemento:", elementHtml);

    // Obter todas as classes dos elementos filhos
    const classes = new Set();
    element.querySelectorAll("*").forEach(child => {
        child.classList.forEach(cls => classes.add(cls));
    });

    // Obter os estilos CSS para cada classe
    const cssStyles = {};
    classes.forEach(cls => {
        cssStyles[cls] = getCssOfClass(cls);
    });

    // Converter cssStyles para string CSS
    const cssString = cssStylesToString(cssStyles);
    console.log("Definições CSS das classes:");
    console.log(cssString);

    return { html: elementHtml, css: cssString };
}

// Exemplo de uso
const result = getElementHtmlAndCss("main");
console.log(result.html);
console.log(result.css);

Quando você executar este código, ele irá:

  1. Obter o HTML do elemento especificado.
  2. Obter as classes dos elementos filhos e seus estilos CSS.
  3. Converter os estilos CSS em uma string formatada.
  4. Imprimir tanto o HTML quanto as definições CSS no console.
